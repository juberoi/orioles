        let score = 0;

        let correctAnswer = 0;

        let currentOperation = "";


        function getRandomPlayer() {

            return players[Math.floor(Math.random() * players.length)];

        }


        function generateQuestion() {

            const player1 = getRandomPlayer();

            const player2 = getRandomPlayer();

            

            if (Math.random() > 0.5) {

                // Multiplication

                correctAnswer = player1.number * player2.number;

                currentOperation = `Multiply ${player1.name} (#${player1.number}) x ${player2.name} (#${player2.number})`;

            } else {

                // Division (Ensure clean division)

                let larger = player1.number > player2.number ? player1 : player2;

                let smaller = player1.number > player2.number ? player2 : player1;

                correctAnswer = larger.number % smaller.number === 0 ? larger.number / smaller.number : Math.floor(larger.number / smaller.number);

                currentOperation = `Divide ${larger.name} (#${larger.number}) ÷ ${smaller.name} (#${smaller.number}) (rounded down if needed)`;

            }


            document.getElementById("question").innerText = currentOperation;

            document.getElementById("result").innerText = "";

            document.getElementById("answer").value = "";

        }


        function checkAnswer() {

            const userAnswer = parseInt(document.getElementById("answer").value);

            if (userAnswer === correctAnswer) {

                document.getElementById("result").innerText = "Correct! You earned a point!";

                score++;

            } else {

                document.getElementById("result").innerText = `Incorrect. The correct answer was ${correctAnswer}.`;

            }

            document.getElementById("score").innerText = `Score: ${score}`;

        }


        generateQuestion();
